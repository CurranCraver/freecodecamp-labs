** start of script.js **

const input = document.querySelector('#markdown-input')
const htmlOutput = document.querySelector('#html-output')
const preview = document.querySelector('#preview')
function convertMarkdown() {
let rawOutput = input.value

   rawOutput = rawOutput.replace(/^#{1}\s(.+)/gm, '<h1>$1</h1>');
   rawOutput = rawOutput.replace(/^#{2}\s(.+)/gm, '<h2>$1</h2>');
   rawOutput = rawOutput.replace(/^#{3}\s(.+)/gm, '<h3>$1</h3>');
   rawOutput = rawOutput.replace(/(\*\*|__)(.+?)\1/gm, '<strong>$2</strong>');
   rawOutput = rawOutput.replace(/(\*|_)(.+?)\1/gm, '<em>$2</em>');
   rawOutput = rawOutput.replace(/!\[(.+?)\]\((.+)\)/igm, '<img src="$2" alt="$1">');
   rawOutput = rawOutput.replace(/\[(.+?)\]\((.+?)\)/g, '<a href="$2" target="_blank">$1</a>')
   rawOutput = rawOutput.replace(/^\s*>\s+(.+)/gm, '<blockquote>$1</blockquote>')
  return rawOutput

}


input.addEventListener('input', ()=> {
  const inputText = input.value
  
  

  htmlOutput.textContent = inputText.replace(inputText, convertMarkdown) 
  preview.innerHTML = htmlOutput.textContent


  
 

})
