** start of script.js **

const checkBtn = document.querySelector('#check-btn')
const input = document.querySelector("#text-input")
const result = document.querySelector('#result')
const regex = /^[\da-z]*$/i

checkBtn.addEventListener('click',()=> {
  let text = input.value

  console.log(text)

  if (text === '' || text.trim() === '') {
    alert("Please input a value")
    return
  }
  
  let testText = text.split('').filter(char => regex.test(char)).join('').toLowerCase()
  console.log(testText)

  let reverseText = testText.split('').reverse().join('')

  console.log(reverseText)
 
  if (testText === reverseText) {
    result.classList.add('show')
    result.innerHTML = `${text} is a palindrome`
  console.log(text)}
  else {
    result.classList.add('show')
    result.innerHTML = `${text} is not a palindrome.`
  }
})
