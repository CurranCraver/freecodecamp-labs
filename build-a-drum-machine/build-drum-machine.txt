** start of script.js **

const padBank = document.querySelector('#pad-bank')
const drumBtn = document.querySelectorAll('.drum-pad')
const display = document.querySelector('#display')


drumBtn.forEach(btn => {
  btn.addEventListener('click', ()=> {
    console.log('Button Clicked!')
    let audio = btn.querySelector('.clip')
    console.log(audio)
      playSound(audio, btn)
  })

})

document.addEventListener('keydown', (e)=> {
  const key= e.key.toUpperCase()
  const sound = document.querySelector(`#${key}`)
  const padBtn = sound.parentElement
  
  const drumKeys = /[qweasdzxc]/i

  if (key.match(drumKeys)) {
    playSound(sound, padBtn)
  sound.classList.add('active')
  }

})

function playSound(sound, highlight) {
  sound.currentTime = 0;
  sound.play()


  highlight.classList.add('active')
  console.log(highlight.id)
  display.textContent = highlight.id
  sound.addEventListener('ended', ()=> {
    highlight.classList.remove('active')
    display.textContent = ''
  })
}
