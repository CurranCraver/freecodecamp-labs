function translatePigLatin(string) {
  const consonant = /^([^aeiou]+)/
  const matches = string.match(consonant)
  let piglatin
  if (matches) {
    piglatin = string.replace(/^([^aeiou]+)(\w*)/, '$2$1ay')

  }
  else {
    piglatin = string + 'way'
  }
  return piglatin
}
