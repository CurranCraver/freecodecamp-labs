function titleCase(string) {
  let oldString = string.toLowerCase()
  let newString = ''
  for (let i = 0; i < oldString.length; i++) {
    if (i === 0) {
      newString = oldString.slice(i, i+1).toUpperCase()
      continue
    }
    if (newString.slice(i-1, i) === " ") {
      newString += oldString.slice(i, i + 1).toUpperCase()   
        } else {
          newString += oldString.slice(i, i+1)
        }
  }return newString
}
